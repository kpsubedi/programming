# Download large file in Python with beautiful progress bar

  import requests
  import tqdm


  def download(url: str, filename: str):
      with open(filename, 'wb') as f:
          with requests.get(url, stream=True) as r:
              r.raise_for_status()
              total = int(r.headers.get('content-length', 0))

              tqdm_params = {
                  'desc': url,
                  'total': total,
                  'miniters': 1, 
                  'unit': 'B',
                  'unit_scale': True,
                  'unit_divisor': 1024,
                }
